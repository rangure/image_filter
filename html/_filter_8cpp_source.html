50void Filter::filter_grayscale(Image *img)
51{
52 int channel = img->get_channel();
53 int width = img->get_width();
54 int height = img->get_height();
55 const unsigned char *data = img->get_data();
56 // Check if image already has 2 or less colour channels (if so dont need to apply the filter.)
57 if (channel <= 2)
58 {
59 return;
60 }
61 else
62 {
63 // Change to grayscale by taking and average of the RGB channels to make it 1 channel
64 int channel_new = channel - 2;
65 // create new container for grayscale image
66 unsigned char *newimg = new unsigned char[width * height * channel_new];
67 double avg;
68 int idx;
69 int cur_idx;
70 for (int i = 0; i < height; i++)
71 {
72 for (int j = 0; j < width; j++)
73 {
74 // for each pixel, take the average of of the RGB channels with the formula below
75 idx = img->pixel_index(i, j, 0);
76 // Average = 0.299 * R + 0.587 * G + 0.114 * B
77 avg = 0.299 * data[idx] + 0.587 * data[idx + 1] + 0.114 * data[idx + 2];
78 cur_idx = i * width * channel_new + j * channel_new;
79 newimg[cur_idx] = avg;
80 // has channels representing transparency
81 if (channel_new == 2)
82 {
83 // copies original image to the second channel of the new image
84 newimg[cur_idx + 1] = data[idx + 3];
85 }
86 }
87 }
88 // update image
89 img->replace_data(newimg);
90 img->set_channel(channel_new);
91 }
92}

201void Filter::filter_histogram_equalization(Image *img)
202{
203
204 int channel = img->get_channel();
205 int width = img->get_width();
206 int height = img->get_height();
207 // Creating a copy of pointer to image data(to reduce function call)
208 const unsigned char *data = img->get_data();
209 // Checking if the image is greyscale/single color.
210 if (channel <= 2)
211 {
212 std::vector<int> intensity_dist(256, 0);
213
214 for (int j = 0; j < width * height * channel; ++j)
215 {
216 uint8_t pixel = static_cast<uint8_t>(data[j]);
217 // Pixel value distribution
218 intensity_dist[pixel]++;
219 }
220 // Cummulative distribution function
221 std::vector<int> cdf(256, 0);
222 int sum = 0;
223 for (int i = 0; i < 256; ++i)
224 {
225 sum += intensity_dist[i];
226 cdf[i] = sum;
227 }
228 // Normalize CDF to [0, 255]
229 int min_cdf = cdf[0];
230 int max_cdf = cdf[255];
231
232 for (int i = 0; i < 256; ++i)
233 {
234 cdf[i] = static_cast<int>((cdf[i] - min_cdf) * 255.0 / (max_cdf - min_cdf));
235 }
236
237 unsigned char *equalised = new unsigned char[width * height * channel];
238 // Mapping pixel intensities to new values using CDF
239 for (int i = 0; i < width * height * channel; ++i)
240 {
241 uint8_t pixel = data[i];
242 equalised[i] = static_cast<unsigned char>(cdf[pixel]);
243 }
244 img->replace_data(equalised);
245 }
246 else
247 {
248 std::cout << "This implementation is for grayscale only. Use contrast enhancement or convert this image into grey scale first" << std::endl;
249 }

449/*Function computes and returns a gaussian kernel(2D matrix of values which is used to weigh)
450 neighbour pixels */
451double *Filter::gaussian_kernel(int kernel, double sig)
452{
453 // create array to store gaussian kernel
454 double *gkern = new double[kernel * kernel];
455 // temporary vector used to store values for the gaussian kernel
456 double *temp_vec = new double[kernel];
457 double sum_val = 0;
458 // loop over row
459 for (int i = 0; i < kernel; i++)
460 {
461 // set value to distance between current row and center or kernel
462 // sig = standard deviation = width of curve
463 temp_vec[i] = gaussian_function(i - kernel / 2, sig);
464 sum_val += temp_vec[i];
465 }
466 // find value of gaussian kernel- scaling by the square of sum of gkern
467 sum_val = sum_val * sum_val;
468 for (int i = 0; i < kernel; i++)
469 {
470 for (int j = 0; j < kernel; j++)
471 {
472 gkern[i * kernel + j] = temp_vec[i] * temp_vec[j] / sum_val;
473 }
474 }
475 delete[] temp_vec;
476 return gkern;
477}
478double *Filter::gaussian_vec(int kernel, double sig)
479{
480 double *temp_vec = new double[kernel];
481 double sum_val = 0;
482 for (int i = 0; i < kernel; i++)
483 {
484 temp_vec[i] = gaussian_function(i - kernel / 2, sig);
485 sum_val += temp_vec[i];
486 }
487 for (int i = 0; i < kernel; i++)
488 {
489 temp_vec[i] = temp_vec[i] / sum_val;
490 }

548void Filter::apply_kernel_det(Image *img, int *kernelx, int *kernely, int l)
549{
550 // load data in
551 int channel = img->get_channel();
552 int width = img->get_width();
553 int height = img->get_height();
554 const unsigned char *data = img->get_data();
555 // get start and end coordinates for x,y loops
556 int loopx_start, loopx_end, loopy_start, loopy_end;
557 loopx_start = l / 2;
558 loopx_end = width - l / 2;
559 loopy_start = l / 2;
560 loopy_end = height - l / 2;
561 // temp image buffer to store data
562 double *temp_img = new double[width * height];
563 // max of filtered value
564 double mymax = 0;
565 int idx, val, center_idx, base_x, base_y;
566 double res, gx, gy;
567 // loop over each pixel in the image
568 for (int i = loopy_start; i < loopy_end; i++)
569 {
570 for (int j = loopx_start; j < loopx_end; j++)
571 {
572 base_y = i - l / 2;
573 base_x = j - l / 2;
574 // gradients in x and y
575 gx = 0;
576 gy = 0;
577 // loop over pixels in the kernel
578 for (int k1 = 0; k1 < l; k1++)
579 {
580 for (int k2 = 0; k2 < l; k2++)
581 {
582 idx = img->pixel_index(base_y + k1, base_x + k2, 0);
583 val = 0;
584 // check if the pixel is within the image boundary
585 if (idx != -1)
586 {
587 val = data[idx];
588 }
589 // add the weighted pixel value to the gradient
590 gx += val * kernelx[k1 * l + k2];
591 gy += val * kernely[k2 * l + k1];
592 }
593 }
594 // find the center pixel index on the original image
595 center_idx = img->pixel_index(i, j, 0);
596 center_idx = i * width + j;
597 // find and store mag of gradient vector in the temp buffer
598 res = sqrt(pow(gx, 2) + pow(gy, 2));
599 temp_img[center_idx] = res;
600 if (res > mymax)
601 // update the max value if new res is larger
602 mymax = res;
603 }
604 }
605 // loop over all pixels and update the values
606 for (int i = loopy_start; i < loopy_end; i++)
607 {
608 for (int j = loopx_start; j < loopx_end; j++)
609 {
610 idx = img->pixel_index(i, j, 0);
611 img->set_data_idx(idx, (int)(temp_img[i * width + j] / mymax * 255 + 0.5));
612 }
613 }
614 // clear memeory
615 delete[] temp_img;
616}
